        function toggleMedicalDetail(detailId) {
            const radioYes = document.querySelector(`input[name="${detailId.replace('_details', '')}"]:checked`);
            const detailField = document.getElementById(detailId);
            
            if (radioYes && radioYes.value === 'Ja') {

        // Formular Daten sammeln
        function collectFormData() {
            const formData = new FormData(document.getElementById('memberForm'));
            const data = {};

            // Normale Felder
            for (let [key, value] of formData.entries()) {
                if (data[key]) {
                    if (Array.isArray(data[key])) {
                        data[key].push(value);
                    } else {
                        data[key] = [data[key], value];
                    }
                } else {
                    data[key] = value;
                }
            }

            // Spezielle Arrays f√ºr Mehrfachauswahl
            const aufmerksamkeit = [];
            const trainingsziele = [];
            
            document.querySelectorAll('input[name="aufmerksamkeit"]:checked').forEach(cb => {
                aufmerksamkeit.push(cb.value);
            });
            
            document.querySelectorAll('input[name="trainingsziele"]:checked').forEach(cb => {
                trainingsziele.push(cb.value);
            });

            data.aufmerksamkeit = aufmerksamkeit.join(', ');
            data.trainingsziele = trainingsziele.join(', ');

            return data;
        }

        // Error/Success Messages
        function showMessage(message, type = 'error') {
            const container = document.getElementById(type === 'error' ? 'error-container' : 'success-container');
            container.innerHTML = `
                <div class="${type === 'error' ? 'error-message' : 'success-message'}">
                    <i class="fas fa-${type === 'error' ? 'exclamation-triangle' : 'check-circle'} mr-2"></i>
                    ${message}
                </div>
            `;
            container.scrollIntoView({ behavior: 'smooth', block: 'center' });
        }

        function clearMessages() {
            document.getElementById('error-container').innerHTML = '';
            document.getElementById('success-container').innerHTML = '';
        }

                showMessage(
                    'Verbindung fehlgeschlagen. Bitte versuche es erneut oder kontaktiere uns.',
                    'error'
                );
            } finally {
